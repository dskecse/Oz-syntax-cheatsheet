\begin{longtable}{l r}
\toprule[0.2em]
\multicolumn{1}{l}{\textbf{Operators}} & \textbf{Meaning}\\
\midrule


 %%%%% 
\multicolumn{2}{c}{\textbf{Basic statements}}\\

\begin{oz}
Var = ...
\end{oz}
&variable assignment\\

\begin{oz}
declare Var 
\end{oz}
&global declaration of Var\\
 
\begin{oz}
local Var in 
	...
end
\end{oz}
&local declaration\\
 
 
\begin{oz}
fun {FunName Arg1 ... ArgN}
	...
end
\end{oz}
&function declaration\\
 
\begin{oz}
proc {ProcName Arg1 ... ArgN}
	...
end
\end{oz}
&procedure declaration\\

\begin{oz}
if Condition then
	...
elseif Condition2 then 
	...
else
	...
end
\end{oz}
&if \dots else if \dots else \dots \\
 
 
\begin{oz}
case Var of Pattern_1 then ...
[] Pattern_2 then ...
else ...
end
\end{oz}
&pattern matching \\[0.4em]



%%%%%
\multicolumn{2}{c}{\textbf{Booleans expressions and operators}}\\
\begin{oz}
false 
\end{oz}
&\\
 
\begin{oz}
true
\end{oz}
&\\
 
\begin{oz}
andthen
\end{oz}
&logical and \\
 
\begin{oz}
orelse
\end{oz}
&logical or\\
 
\begin{oz}
==
\end{oz}
&logical equality\\
 
\begin{oz}
\=
\end{oz}
&logical inequality\\
 
\begin{oz}
{Not [Your Expression]}
\end{oz}
&logical not\\[0.4em]
 
 
%%%%% 
\multicolumn{2}{c}{\textbf{Comparison operators}}\\

%%%%% 
\multicolumn{2}{c}{\textbf{Arithmetic operators}}\\
\begin{oz}
+
\end{oz}
&addition\\
 
\begin{oz}
-

&subtraction\\
 
\begin{oz}
*
\end{oz}
&multiplication\\
 
\begin{oz}

mod

 %%%%% 
\multicolumn{2}{c}{\textbf{Data structures}}\\

\begin{oz}
X = {NewCell Y}
\end{oz}
&creates a new cell (multiple assignment variable) \\

\begin{oz}
@X
\end{oz}
& returns the cell's current content\\

\begin{oz}
X := Z
\end{oz}
&changes the content of the cell \\[0.4em]


 %%%%%

\begin{oz}

\end{oz}

\begin{oz}
try ... catch X then ... end
\end{oz}
&catches a raised exception\\
